// скачать образ ubuntu и присвоить тэг latest
docker pull ubuntu:latest

// Посмотреть образы
docker images

// запустить образ
docker run -ti ubuntu:latest

// информация о запущенном контейнере
docker ps

// информация о всех контейнерах запущенных и не запущенных
docker ps -a

// Команда в контейнере 
// Имя ОС
uname -s

// Обновление репозиториев
apt update

// устанавливаем компилятор и текстовый редактор
apt install g++ nano
